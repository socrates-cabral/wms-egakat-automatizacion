**Goal:** Agregar wake word "Hey Jarvis" (Porcupine) y STT local offline (faster-whisper) reemplazando la key pública de Chromium.

**Architecture:** WakeWordDetector corre en thread daemon alimentando Porcupine con frames de sounddevice 16kHz/512 samples. Al detectar, llama `harness.trigger()` igual que Win+J. **Tech Stack:** Python 3.14, pvporcupine>=3.0, faster-whisper>=1.0, sounddevice, numpy, PyQt6 (sin cambios en UI)

```python
import sys
try:
    sys.stdout.reconfigure(encoding="utf-8")
except (AttributeError, ValueError):
    pass

import logging
import numpy as np

from jarvis.config import STT_MODEL, STT_LANGUAGE

logger = logging.getLogger("jarvis.stt")

_model = None


def _get_model():
    global _model
    if _model is None:
        from faster_whisper import WhisperModel
        logger.info("Cargando faster-whisper '%s' (primera vez ~2s)...", STT_MODEL)
        _model = WhisperModel(STT_MODEL, device="cpu", compute_type="int8")
        logger.info("faster-whisper listo.")
    return _model


def transcribe(pcm_int16: bytes, samplerate: int = 16000) -> str:
    """Transcribe PCM int16 bytes a texto en español. Retorna '' si falla o sin habla.

    Si samplerate != 16000, resamplea mediante interpolación lineal antes de enviar
    a Whisper (que requiere 16kHz).
    """
    try:
        model = _get_model()
        audio = np.frombuffer(pcm_int16, dtype="<i2").astype("float32") / 32768.0
        if samplerate != 16000 and samplerate > 0:
            n_out = int(len(audio) * 16000 / samplerate)
            if n_out > 0:
                audio = np.interp(
                    np.linspace(0, len(audio) - 1, n_out),
                    np.arange(len(audio)),
                    audio,
                ).astype("float32")
        segments, _ = model.transcribe(
            audio,
            language=STT_LANGUAGE,
            beam_size=1,
            vad_filter=True,
        )
        text = " ".join(s.text.strip() for s in segments).strip()
        if text:
            logger.info("STT: '%s'", text)
        return text
    except Exception as e:
        logger.error("STT error: %s", e)
        return ""
```

```python
"""Wake word detector usando Porcupine (pvporcupine).

Uso:
    from jarvis.wakeword import WakeWordDetector
    detector = WakeWordDetector(callback=harness.trigger)
    detector.start(access_key, keyword_path, sensitivity=0.5)
    # ... al salir:
    detector.stop()
"""
import sys
try:
    sys.stdout.reconfigure(encoding="utf-8")
except (AttributeError, ValueError):
    pass

import logging
import threading
from typing import Callable

logger = logging.getLogger("jarvis.wakeword")


class WakeWordDetector:
    """Detecta 'Hey Jarvis' usando Porcupine en un thread daemon.

    Si pvporcupine no está instalado o la key/path son inválidos, start()
    retorna False y la app sigue funcionando solo con Win+J.
    """

    def __init__(self, callback: Callable[[], None]):
        self._callback   = callback
        self._stop_event = threading.Event()
        self._thread: threading.Thread | None = None
        self._running    = False

    def is_running(self) -> bool:
        return self._running

    def start(self, access_key: str, keyword_path: str,
              sensitivity: float = 0.5) -> bool:
        """Inicia el detector. Retorna True si arrancó correctamente."""
        try:
            import pvporcupine
            porcupine = pvporcupine.create(
                access_key=access_key,
                keyword_paths=[keyword_path],
                sensitivities=[sensitivity],
            )
        except Exception as e:
            logger.warning("Porcupine init falló: %s — wake word desactivado", e)
            return False

        self._stop_event.clear()
        self._running = True
        self._thread = threading.Thread(
            target=self._loop,
            args=(porcupine,),
            daemon=True,
            name="wakeword-porcupine",
        )
        self._thread.start()
        logger.info("Wake word detector activo (sensitivity=%.1f)", sensitivity)
        return True

    def stop(self) -> None:
        """Detiene el detector limpiamente."""
        self._stop_event.set()
        if self._thread is not None:
            self._thread.join(timeout=2.0)
        self._running = False

    def _loop(self, porcupine) -> None:
        import sounddevice as sd
        frame_size = porcupine.frame_length   # típicamente 512 samples @ 16kHz = 32ms
        try:
            with sd.InputStream(
                samplerate=16000,
                channels=1,
                dtype="int16",
                blocksize=frame_size,
            ) as stream:
                while not self._stop_event.is_set():
                    frame, _ = stream.read(frame_size)
                    pcm = frame[:, 0].tolist()
                    idx = porcupine.process(pcm)
                    if idx >= 0:
                        logger.info("Wake word detectado — activando ciclo STT")
                        self._callback()
        except Exception as e:
            logger.error("Wake word loop error: %s", e)
        finally:
            try:
                porcupine.delete()
            except Exception:
                pass
            self._running = False
```

    def start(self) -> None:
        """Inicializa agente, restaura timers, arranca wake word. Llamar una vez al arrancar."""
        context = self._memory.load_context()
        self._agent = Agent(memory_context=context)
        self._bridge.tts_cancel_requested.connect(voice.cancel_tts)

        from jarvis.tools import restore_timers
        for msg in restore_timers():
            logger.info("Timer restaurado: %s", msg)

        # Wake word detector — falla silenciosamente si Porcupine no está configurado
        from jarvis.wakeword import WakeWordDetector
        from jarvis.config import (
            PORCUPINE_ACCESS_KEY, WAKE_WORD_PATH,
            WAKE_WORD_SENSITIVITY,
        )
        self._wakeword = WakeWordDetector(callback=self.trigger)
        if PORCUPINE_ACCESS_KEY and WAKE_WORD_PATH.exists():
            ok = self._wakeword.start(
                PORCUPINE_ACCESS_KEY,
                str(WAKE_WORD_PATH),
                WAKE_WORD_SENSITIVITY,
            )
            if not ok:
                logger.warning("Wake word no pudo iniciar — solo Win+J disponible.")
        else:
            missing = []
            if not PORCUPINE_ACCESS_KEY:
                missing.append("PORCUPINE_ACCESS_KEY en .env")
            if not WAKE_WORD_PATH.exists():
                missing.append(f"jarvis_windows.ppn en {WAKE_WORD_PATH.parent}")
            logger.warning(
                "Wake word desactivado — falta: %s", ", ".join(missing)
            )

        logger.info("Harness inicializado. Contexto de memoria cargado.")

    def stop_wakeword(self) -> None:
        """Detiene el wake word detector. Llamar en app.aboutToQuit."""
        if hasattr(self, "_wakeword"):
            self._wakeword.stop()
            logger.info("Wake word detector detenido.")
