```python
def _record_sounddevice(duration_max: int, device: int) -> bytes | None:
    """Graba con sounddevice + webrtcvad: corta al detectar silencio post-voz."""
    try:
        import sounddevice as sd
        import numpy as np

        try:
            import webrtcvad
            vad = webrtcvad.Vad(_VAD_AGG)
            use_vad = True
        except ImportError:
            vad = None
            use_vad = False
            logger.warning("webrtcvad no disponible — grabando %ds fijos", duration_max)

        SILENCE_CHUNKS = int(_VAD_SILENCE_S * 1000 / _VAD_CHUNK_MS)  # ~27 chunks
        MAX_CHUNKS     = int(_VAD_MAX_S   * 1000 / _VAD_CHUNK_MS)

        frames        = []
        silence_count = 0
        speech_found  = False

        with sd.InputStream(samplerate=_VAD_RATE, channels=1, dtype='int16',
                            device=device, blocksize=_VAD_CHUNK_SAMP) as stream:
            for _ in range(MAX_CHUNKS):
                if _cancel_tts.is_set():   # reusar el mismo evento de cancelación
                    return None
                chunk, _ = stream.read(_VAD_CHUNK_SAMP)
                pcm = chunk[:, 0].tobytes()
                frames.append(pcm)

                if use_vad:
                    is_speech = vad.is_speech(pcm, _VAD_RATE)
                else:
                    # Fallback energía RMS
                    rms = (sum(v*v for v in chunk[:, 0]) / len(chunk)) ** 0.5
                    is_speech = rms > 300

                if is_speech:
                    speech_found  = True
                    silence_count = 0
                elif speech_found:
                    silence_count += 1
                    if silence_count >= SILENCE_CHUNKS:
                        break

        if not speech_found or not frames:
            return None

        # Unir chunks, boost x25, devolver a 16kHz mono
        import numpy as np
        raw  = np.frombuffer(b"".join(frames), dtype='<i2')
        boosted = np.clip(raw.astype('int32') * _MIC_BOOST, -32768, 32767).astype('int16')
        peak = int(abs(boosted).max())
        logger.info("sounddevice VAD: %d frames, peak=%d (%.0f%%)",
                    len(frames), peak, peak/32767*100)
        return boosted.tobytes()

    except Exception as e:
        logger.warning("sounddevice VAD falló: %s", e)
        return None
```

```python
def _play_audio(path: str) -> None:
    """Reproduce MP3 cancelable via _cancel_tts. Intenta playsound3, fallback PS."""
    _cancel_tts.clear()
    try:
        from playsound3 import playsound
        # playsound es bloqueante; correr en thread y monitorear cancel
        done = _threading.Event()
        def _play():
            try:
                playsound(path)
            except Exception:
                pass
            finally:
                done.set()
        t = _threading.Thread(target=_play, daemon=True)
        t.start()
        while not done.wait(timeout=0.05):
            if _cancel_tts.is_set():
                return
        return
    except Exception:
        pass

    # Fallback PowerShell — proceso externo cancelable
    try:
        safe = str(Path(path).resolve()).replace("'", "''")
        ps = (
            "Add-Type -AssemblyName presentationCore; "
            f"$p = New-Object System.Windows.Media.MediaPlayer; "
            f"$p.Open([uri]'{safe}'); $p.Play(); "
            "Start-Sleep -Milliseconds 500; "
            "while ($p.NaturalDuration.HasTimeSpan -eq $false) { Start-Sleep -Milliseconds 100 }; "
            "Start-Sleep -Seconds ([int]$p.NaturalDuration.TimeSpan.TotalSeconds + 1)"
        )
        proc = subprocess.Popen(["powershell", "-Command", ps], 
                                stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
        _tts_proc[0] = proc
        while proc.poll() is None:
            if _cancel_tts.is_set():
                proc.terminate()
                break
            _threading.Event().wait(0.05)
        _tts_proc[0] = None
    except Exception as e:
        logger.error("Audio playback error: %s", e)
```

```python
import time as _time
import uuid as _uuid

_TIMERS_PATH = Path(__file__).parent / "data" / "timers.json"

def _load_timers() -> dict:
    if _TIMERS_PATH.exists():
        try:
            return json.loads(_TIMERS_PATH.read_text(encoding="utf-8"))
        except Exception:
            return {}
    return {}

def _save_timer(tid: str, vence_en: float, mensaje: str) -> None:
    _TIMERS_PATH.parent.mkdir(parents=True, exist_ok=True)
    timers = _load_timers()
    timers[tid] = {"vence_en": vence_en, "mensaje": mensaje}
    _TIMERS_PATH.write_text(json.dumps(timers, indent=2), encoding="utf-8")

def _clear_timer(tid: str) -> None:
    timers = _load_timers()
    timers.pop(tid, None)
    _TIMERS_PATH.write_text(json.dumps(timers, indent=2), encoding="utf-8")

def restore_pending_timers() -> int:
    """Reprogramar timers persistidos al arrancar. Retorna cantidad reprogramada."""
    timers = _load_timers()
    now    = _time.time()
    count  = 0
    for tid, data in list(timers.items()):
        remaining = data["vence_en"] - now
        if remaining > 0:
            msg = data["mensaje"]
            def _alert(t=tid, m=msg):
                subprocess.run(
                    ["powershell", "-NonInteractive", "-Command",
                     "param([string]$m) Add-Type -AssemblyName System.Windows.Forms; "
                     "[System.Windows.Forms.MessageBox]::Show($m,'J.A.R.V.I.S.')",
                     "-m", m],
                    capture_output=True, timeout=30,
                )
                _clear_timer(t)
            t = threading.Timer(remaining, _alert)
            t.daemon = True
            t.start()
            count += 1
        else:
            _clear_timer(tid)
    return count
```

```python
def _calc_pill_height(self, text: str) -> int:
    """Calcula altura del pill según longitud del texto de respuesta."""
    CHARS_PER_LINE = 48   # ~chars por línea a Consolas 9px en 360px de ancho
    LINE_H         = 18   # px por línea
    STATUS_H       = 52   # altura fija del status row
    PADDING        = 24   # padding top + bottom
    PILL_MAX_H     = 300

    lines = max(1, len(text) // CHARS_PER_LINE + text.count("\n") + 1)
    height = STATUS_H + PADDING + lines * LINE_H
    return min(height, PILL_MAX_H)
```
